## Configuration

Make sure to add `ThemeProvider` into your project's main layout right as the first child of `<body>`.

```tsx showLineNumbers title="app/layout.tsx" caption="add your layout root"
import { ThemeProvider, ThemeStateHidden } from "@/config/themes";

export default function RootLayout({ children }: Readonly<{ children: React.ReactNode }>) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <ThemeProvider>
          {children}
          <ThemeStateHidden />
        </ThemeProvider>
      </body>
    </html>
  );
}
```
