## Installation

Install the required peer dependency:

```bash
npm install sonner
```

> Oeri UI's `Toaster` component is powered by [`sonner`](https://sonner.emilkowal.ski/), a minimal toast notification library with great accessibility and performance.

---

## Configuration

Make sure to include the `<Toaster />` component inside your application layout â€” typically after your `ThemeProvider`.

```tsx title="app/layout.tsx" showLineNumbers caption="Add the Toaster in your root layout"
import { ThemeProvider } from "@/config/themes";
import { Toaster } from "@/ui/toaster";

export default function RootLayout({ children }: Readonly<{ children: React.ReactNode }>) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <ThemeProvider>
          {children}
          <Toaster />
        </ThemeProvider>
      </body>
    </html>
  );
}
```
