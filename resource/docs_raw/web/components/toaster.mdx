## Configuration

Make sure to add `<Toaster />` to your project's main layout inline with the children of `<ThemeProvider>`.

```tsx showLineNumbers title="app/layout.tsx" caption="add your layout root"
import { ThemeProvider } from "@/config/themes";
import { Toaster } from "@/ui/toaster";

export default function RootLayout({ children }: Readonly<{ children: React.ReactNode }>) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <ThemeProvider>
          {children}
          <Toaster />
        </ThemeProvider>
      </body>
    </html>
  );
}
```
